# TransformX-Data-API

installation:

directory: 
mkdir data-curation-api
cd data-curation-api

frontend: 

npx create-react-app frontend
cd frontend
npm install axios
npm run build


backend:

virtual environment:
python -m venv venv

flask:
pip install flask pandas
pip install fastapi uvicorn pandas
pip install flask-cors

Create App.py and start writing your APIs.
